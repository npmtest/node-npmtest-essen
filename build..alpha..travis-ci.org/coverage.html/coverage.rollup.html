module.exports = class EORM {
  static init() {
    mongoose.Promise = global.Promise
    EORM.models = {}
    return new Promise((resolve, reject) => {
      EORM.loadModels(err => {
        if (err) reject(`ORM models error ${err}`)
        EORM.initGlobalModels(err => {
          if (err) reject('ORM models error', err)
          log.debug('ORM models loaded')
          return resolve()
        })
      })
    })
  }
  static loadModels(cb) {
    const models_path = path.join(essen.config.path, 'models')
    fs.readdir(models_path, (err, files_names) => {
      async.each(files_names, (file_name, next) => {
        const model = {
          file_name,
          name: file_name.split('Model')[0],
          file_path: path.join(models_path, file_name),          
        }
        EORM.loadModel(model, err => {
          if (err) return cb(err)
          return next()
        })
      }, cb)
    })
  }
  static loadModel(model, cb) {
    const schema_object = require(model.file_path)
    model.attributes = schema_object.attributes || {}
    model.methods = schema_object.methods || {}
    model.statics = schema_object.statics || {}
    model.pre = schema_object.pre || {}
    model.post = schema_object.post || {}
    const schema = new mongoose.Schema(model.attributes)
    for (let method in model.statics) {
      schema.statics[method] = model.statics[method]
    }
    for (let method in model.methods) {
      schema.methods[method] = model.methods[method]
    }
    for (let event in model.pre) {
      schema.pre(event, model.pre[event])
    }
    for (let event in model.post) {
      schema.post(event, model.post[event])
    }
    EORM.models[model.name] = mongoose.model(model.name, schema)
    return cb()
  }
  static initGlobalModels(cb) {
    const config = essen.config.db
    if (config.login && config.password) {
      mongoose.connect(`mongodb://${config.login}:${config.password}@${config.host}:${config.port}/${config.name}`)
    } else {
      mongoose.connect(`mongodb://${config.host}:${config.port}/${config.name}`)      
    }
    for (let model_name in EORM.models) {
      global[model_name] = EORM.models[model_name]
    }
    return cb()
  }
}

const fs = require('fs')
const path = require('path')
const async = require('async')
 
module.exports = class ERouter {
  static init() {
    return new Promise((resolve, reject) => {
      ERouter.controllers = {}
      ERouter.loadRoutes(routes => {
        ERouter.listenRoutes('/', routes, err => {
          if (err) reject(err)
          log.debug('ERouter inited')
          resolve()
        })
      })
    })
  }
  static loadRoutes(cb) {
    const routes = require(path.join(essen.config.path, 'router/router.js'))
    return cb(routes)
  }
  static listenRoutes(namespace, routes, cb) {
    async.each(routes, (route, next) => {
      ERouter.listenRoute(namespace, route, err => {
        if (err) return next(err)
        return next()
      })
    }, err => {
      return cb(err)
    })
  }
  static listenRoute(namespace, route, cb) {
    if (!route.namespace) {
      let controller_path = path.join(essen.config.path, 'controllers', namespace, route.controller + '.js')
      let controller
      try {
        controller = require(controller_path)
      } catch(e) {
        console.log(namespace, route)
        return cb(`controller ${route.controller} is not found in ${controller_path}`)
      }
      let final_path = namespace + route.path
      log.verbose(route.controller + ':' + route.action + ' listen to ' + final_path)
      if (!controller[route.action]) {
        return cb(`action '${route.action}' not found in '${route.controller}'`)
      } else {
        essen.server[route.type.toLowerCase()](namespace + route.path, controller[route.action])
      }
      if (route.children) {
        namespace = namespace + route.path + '/'
        ERouter.listenRoutes(namespace, route.children, err => {
          return cb(err)
        })
      } else {
        return cb()
      }
    } else {
      if (route.children) {
        namespace = namespace + route.namespace + '/'
        ERouter.listenRoutes(namespace, route.children, err => {
          return cb(err)
        })
      } else {
        return cb()
      }
    }
  }
}

module.exports = class EServer {
  static start() {
    const server = require('http').createServer(essen.server)
    if (essen.config.socket.enabled) {
      essen.io = require('socket.io')(server)      
    }
    return new Promise((resolve, reject) => {
      server.listen(process.env.PORT || essen.config.server.port, err => {
        if (err) reject(`start server error ${err}`)
        log.info(`${essen.config.server.name} listening on port ${process.env.PORT || essen.config.server.port}`)
        log.info(`using db ${essen.config.db.name} at ${essen.config.db.host}`)
        if (essen.afterStart) essen.afterStart()
        resolve()
      })
    })
  }
}

      EServiceman.loadServices(err => {
        if (err) reject (`services init error ${err}`)
        log.debug('EServiceman loaded')
        return resolve()
      })
    })

  static loadServices(cb) {
    fs.readdir(EServiceman.path, (err, files_names) => {
      async.each(files_names, (file_name, next) => {
        let file_path = path.join(EServiceman.path, file_name)
        let service_name = file_name.split('.')[0]
        global[service_name] = require(file_path)
        return next()
      }, cb)
    })
  }
}
